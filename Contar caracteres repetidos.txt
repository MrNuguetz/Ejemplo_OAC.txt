%include "./pc_io.inc" 

section .data
msg: db 'ingrese un texto: ',0x0

section .bss
aux resb 1
aux2 resb 1



section .text
   
global _start

_start:
mov edx,msg
call puts
call salto


call getch
mov byte[aux],al
mov al,byte[aux]
call putchar
call salto

ciclo:

    call getch
    mov byte[aux2],al
    mov al,byte[aux2]
    cmp al,'*'
    je exit
    call putchar
    jmp ciclo

exit:
mov eax, 1       
mov ebx, 0       
int 80h


 

salto:
    pushad
    mov al,13
    call putchar
    mov al,10
    call putchar
    popad
    ret
