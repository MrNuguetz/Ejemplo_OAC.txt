%include "./pc_io.inc" 

section .data

section .bss


cad resb 10
section .text

global _start
_start:

call ciclo
call salto

mov eax,1
mov ebx,0
int 80h

ciclo:
    mov ecx,0
    mov esi,cad
        ciclocap:
        cmp ecx,100
        je salir
        mov eax,ecx
        miniciclo:
        sub eax,2
        cmp eax,0
        je par 
        cmp eax,1
        je impar
        jmp miniciclo

        jmp ciclocap

        par:
        mov eax,ecx
        call printHex8
        call salto
        inc ecx
        jmp ciclocap

        impar:
        inc ecx
        jmp ciclocap

        salir:
        mov eax,1
        mov ebx,0
        int 80h


;en eax el valor a convertir mostrar en hexadecimal
printHex8:
  pushad
  mov edx, eax
  mov ebx, 0fh
  mov cl, 28
.nxt: shr eax,cl
.msk: and eax,ebx
  cmp al, 9
  jbe .menor
  add al,7
.menor:add al,'0'
  mov byte [esi],al
  inc esi
  mov eax, edx
  cmp cl, 0
  je .print
  sub cl, 4
  cmp cl, 0
  ja .nxt
  je .msk
.print: mov eax, 4
  mov ebx, 1
  sub esi, 8
  mov ecx, esi
  mov edx, 8
  int 80h
  popad
  ret

salto:
    pushad
    mov al,13
    call putchar
    mov al,10
    call putchar
    popad
    ret
